USERS:
	Add New User:
		- Endpoint : /add_user
		- Request Type : POST
		- Request:
			- String : fname
			- String : lname
			- String : email (email is the same as userId)
		- Response:
			- String : userId (-1 on failure)
			- String : error (null on failure)
	Get User:
		- Endpoint : /get_user
		- Request Type : POST
		- Request : 
			- String : email
			userId,zstring:workspaceId,string:authCode
		- Response :
			- String : fname
			- String : lname
			- String : email
			- String : error

STRATEGY:
	Create Strategy
		- Endpoint : /create_strategy
		- Request Type : POST
		- Request:
			- int : workplaceId
			- String : strategyName
			- String : description
			- String : color
		- Response:
			- int : strategyId (id or -1 on error)
			- String : error (null if no error)

	Update Strategy
		- Endpoint : /update_strategy
		- Request Type : POST
		- Request:
			- string : strategyId
			- string : name 
			- String : description
		- Response:
			- bool : success
	
	Get Strategies
		- Endpoint : /get_strategy
		- Request Type : POST
		- Request:
			- int : workspaceId
		- Response:
			- Object [] : myStrategyList (or empty list)
				- int : strategyId
				- String : strategyName
				- String : description
				- String : color

GOALS:
	Create Goal
		- Endpoint : /create_goal
		- Request Type : POST
		- Request:
			- String:strategyId
			- String:goalName
			- String:description
		- Response:
			- String : error
			- String : goalId
